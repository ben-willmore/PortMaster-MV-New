# build the develop branch 
git checkout develop

# OpenGLES is not available on 8.0.6 release, so we use this commit
# git checkout 651348d2a94f6bfa5f7e7f89800908e9e8a4a3e5

 -DBUILD_CROWD_CONTROL=0 -DCMAKE_BUILD_TYPE=Release
